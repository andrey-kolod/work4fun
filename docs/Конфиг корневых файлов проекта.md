// src/types/index.types.ts

/**
 * Конфигурация метрик
 */
export interface MetricsConfig {
  activeUserWindowMs: number;      // окно активности пользователя (мс)
  cleanupIntervalMs: number;        // интервал очистки (мс)
  flushIntervalMs: number;          // интервал сброса (не используется)
}

/**
 * Описание кастомных метрик
 */
export interface CustomMetrics {
  testCounter: {
    name: string;      // app_test_counter
    help: string;      // Help text для Prometheus
  };
  httpRequestsTotal: {
    name: string;                      // app_http_requests_total
    help: string;                       // Help text
    labelNames: readonly ['method', 'endpoint', 'status']; // метки для группировки
  };
  httpRequestDurationSeconds: {
    name: string;                      // app_http_request_duration_seconds
    help: string;                       // Help text
    labelNames: readonly ['method', 'endpoint', 'status']; // метки
    buckets: number[];                   // границы гистограммы (в секундах)
  };
  activeUsers: {
    name: string;      // app_active_users
    help: string;      // Help text
  };
}

/**
 * Тип для глобальной переменной метрик (синглтон)
 */
export interface GlobalMetrics {
  register: any;                      // Prometheus registry
  testCounter: any;                    // тестовый счётчик
  httpRequestTotal: any;                // счётчик HTTP запросов
  httpRequestDuration: any;             // гистограмма длительности
  activeUserGauge: any;                 // gauge активных пользователей
  cleanupCounter: any;                  // счётчик очисток
  activeUsersMap: Map<string, number>;  // карта userId -> timestamp
  cleanupTimer: NodeJS.Timeout;          // таймер очистки
  isInitialized: boolean;                // флаг инициализации
  lastCleanupTime: number;                // время последней очистки
}
🎯 Какие метрики собираются
Метрика	Тип	Описание	Пример
app_test_counter	Counter	Тестовый счётчик для проверки	app_test_counter 42
app_http_requests_total	Counter	Общее количество HTTP запросов	app_http_requests_total{method="GET",endpoint="/api/projects",status="200"} 150
app_http_request_duration_seconds	Histogram	Распределение времени ответа	app_http_request_duration_seconds_bucket{le="0.1",method="GET"} 120
app_active_users	Gauge	Количество активных пользователей за 30 мин	app_active_users 5
app_users_cleanup_total	Counter	Сколько раз чистили список	app_users_cleanup_total{deleted="3"} 1
app_health	Gauge	Статус здоровья метрик	app_health{service="metrics"} 1
